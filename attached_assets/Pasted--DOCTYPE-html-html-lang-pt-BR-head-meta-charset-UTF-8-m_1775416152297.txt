 * Próximos níveis = "lock" até acumular check-ins suficientes.
 * A cada 40 check-ins, o próximo nível é desbloqueado (com unlock antecipado a 36).
 */
function getStatus(idx,inicialIdx,progresso){
  if(idx<inicialIdx) return 'done';           // superado
  if(idx===inicialIdx) return 'open';          // nível de entrada — sempre ativo
  // para níveis além do inicial: verifica check-ins acumulados
  const ciclos=idx-inicialIdx;               // quantos ciclos além do inicial
  const minimo=(ciclos-1)*40+36;             // 36 = 90% de 40 (desbloqueio antecipado)
  return progresso>=minimo ? 'open' : 'lock';
}

// Progresso local dentro do ciclo atual
function getLocalProg(progresso,idx,inicialIdx){
  if(idx<inicialIdx) return 40;
  if(idx===inicialIdx){
    // Antes de qualquer check-in, o nível inicial já está disponível mas com prog=0
    return Math.min(progresso,40);
  }
  const ciclo=idx-inicialIdx;
  const base=(ciclo-1)*40;
  return Math.min(Math.max(progresso-base,0),40);
}

// Índice do nível mais avançado desbloqueado (nível "ativo" atual)
function getNivelAtualIdx(inicialIdx,progresso){
  let idx=inicialIdx;
  for(let i=inicialIdx+1;i<LEVELS.length;i++){
    if(getStatus(i,inicialIdx,progresso)==='open') idx=i;
    else break;
  }
  return idx;
}

// Dias para desbloquear o próximo nível
function diasParaProx(progresso,nivelAtualIdx,inicialIdx){
  const nextIdx=nivelAtualIdx+1;
  if(nextIdx>=LEVELS.length) return 0;
  const ciclos=nextIdx-inicialIdx;
  const minimo=(ciclos-1)*40+36;
  return Math.max(minimo-progresso,0);
}

function playSound(type){
  try{
    const ctx=new(window.AudioContext||window.webkitAudioContext)();
    const o=ctx.createOscillator(),g=ctx.createGain();
    o.connect(g);g.connect(ctx.destination);
    if(type==='ci'){
      o.frequency.setValueAtTime(523,ctx.currentTime);
      o.frequency.setValueAtTime(659,ctx.currentTime+0.1);
      o.frequency.setValueAtTime(784,ctx.currentTime+0.2);
      g.gain.setValueAtTime(0.22,ctx.currentTime);
      g.gain.exponentialRampToValueAtTime(0.001,ctx.currentTime+0.5);
      o.start();o.stop(ctx.currentTime+0.5);
    } else {
      [523,784,1047,1319].forEach((f,i)=>o.frequency.setValueAtTime(f,ctx.currentTime+i*0.15));
      g.gain.setValueAtTime(0.28,ctx.currentTime);
      g.gain.exponentialRampToValueAtTime(0.001,ctx.currentTime+0.9);
      o.start();o.stop(ctx.currentTime+0.9);
    }
  }catch(e){}
}

/* ══ APP ROOT ══ */
function App(){
  const [user,setUser]=useState(()=>db.getUser());
  const [lang,setLang]=useState(()=>db.getLang());
  const [page,setPage]=useState('dashboard');
  const [ciModal,setCiModal]=useState(false);
  const [luModal,setLuModal]=useState(null);

  const handleLang=useCallback((l)=>{setLang(l);db.setLang(l)},[]);

  /* ── CHECK-IN MANUAL: chamado apenas pelo botão no Dashboard ── */
  const handleCheckin=useCallback(()=>{
    if(!user||user.ultimo_checkin===TODAY) return; // guarda duplicidade
    const ii=LEVELS.findIndex(l=>l.id===user.nivel_inicial);
    const prevNi=getNivelAtualIdx(ii,user.progresso);
    const upd={...user,progresso:user.progresso+1,ultimo_checkin:TODAY};
    const newNi=getNivelAtualIdx(ii,upd.progresso);
    db.saveUser(upd);
    db.addCheckin(TODAY); // salva no histórico do calendário
    setUser(upd);
    playSound('ci');
    setCiModal(true);
    if(newNi>prevNi){
      setTimeout(()=>{
        setCiModal(false);
        playSound('lu');
        setLuModal(LEVELS[newNi]);
      },2200);
    }
  },[user]);

  const t=T[lang];

  if(!user) return(
    <div className="app">
      <LoginPage onLogin={u=>{setUser(u)}} lang={lang} setLang={handleLang}/>
    </div>
  );

  const renderPage=()=>{
    switch(page){
      case 'dashboard':  return <DashboardPage user={user} lang={lang} t={t} setLang={handleLang} onCheckin={handleCheckin}/>;
      case 'feed':       return <FeedPage lang={lang} t={t}/>;
      case 'comunidade': return <ComunidadePage user={user} lang={lang} t={t}/>;
      case 'instrucoes': return <InstrucoesPage t={t}/>;
      case 'perfil':     return <PerfilPage user={user} lang={lang} t={t} onLogout={()=>{db.clearUser();setUser(null);}}/>;
      default:           return <DashboardPage user={user} lang={lang} t={t} setLang={handleLang} onCheckin={handleCheckin}/>;
    }
  };

  return(
    <div className="app">
      {ciModal&&<CheckinModal user={user} lang={lang} t={t} onClose={()=>setCiModal(false)}/>}
      {luModal&&<LevelUpModal nivel={luModal} lang={lang} t={t} onClose={()=>setLuModal(null)}/>}
      <div className="fade-in">{renderPage()}</div>
      <BottomNav active={page} setActive={setPage} t={t}/>
    </div>
  );
}

ReactDOM.render(<App/>,document.getElementById('root'));
</script>
